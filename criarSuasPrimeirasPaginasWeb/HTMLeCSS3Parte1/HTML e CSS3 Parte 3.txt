/* sobre a força ds tags, ids, classes etc */

/*a tag tem uma força de 1*/


p{
	color: red; 
}

/*1 + 1 = 2*/
form p{
	color: blue;
}

/*A class tem uma força 10*/
.teste{
	color: yellow;
}

/*1 + 10 = 11*/
p.teste{
	color: orange;
}

/*id tem força 100*/
#teste{
	color: hotpink;
}

/* Resumindo: o mais forte seria o inline, que é direto no html. Vai então da especifidade e poder tomar melhor as decisões */
